        // ========== PAGINATION ==========
        function renderPagination(totalItems, currentPage) {
            const totalPages = Math.ceil(totalItems / itemsPerPage);
            const pagination = document.getElementById('pagination');
            pagination.innerHTML = '';

            if (totalPages <= 1) return;

            // Prev
            const prevBtn = document.createElement('button');
            prevBtn.className = 'page-btn';
            prevBtn.innerHTML = '<i class="fas fa-chevron-left"></i>';
            prevBtn.disabled = currentPage === 1;
            prevBtn.onclick = () => renderDrivers(getFilteredDrivers(), currentPage - 1);
            pagination.appendChild(prevBtn);

            // Pages
            for (let i = 1; i <= totalPages; i++) {
                const btn = document.createElement('button');
                btn.className = `page-btn ${i === currentPage ? 'active' : ''}`;
                btn.textContent = i;
                btn.onclick = () => renderDrivers(getFilteredDrivers(), i);
                pagination.appendChild(btn);
            }

            // Next
            const nextBtn = document.createElement('button');
            nextBtn.className = 'page-btn';
            nextBtn.innerHTML = '<i class="fas fa-chevron-right"></i>';
            nextBtn.disabled = currentPage === totalPages;
            nextBtn.onclick = () => renderDrivers(getFilteredDrivers(), currentPage + 1);
            pagination.appendChild(nextBtn);
        }

        // ========== FILTERS ==========
        function getFilteredDrivers() {
            const status = document.getElementById('statusFilter').value;
            const vehicle = document.getElementById('vehicleFilter').value;
            const zone = document.getElementById('zoneFilter').value;
            const search = document.getElementById('searchInput').value.toLowerCase();

            return drivers.filter(d => {
                const matchStatus = status === 'all' || d.status === status;
                const matchVehicle = vehicle === 'all' || d.type === vehicle;
                const matchZone = zone === 'all' || d.zone === zone;
                const matchSearch = d.name.toLowerCase().includes(search) || 
                                   d.vehicle.toLowerCase().includes(search) ||
                                   d.phone.includes(search);
                return matchStatus && matchVehicle && matchZone && matchSearch;
            });
        }

        function saveDriver(event) {
            event.preventDefault();
            const firstName = document.getElementById('firstName').value;
            const lastName = document.getElementById('lastName').value;
            const name = `${firstName} ${lastName}`;

            const newDriver = {
                id: currentDriverId || Date.now(),
                name: name,
                phone: document.getElementById('phone').value,
                email: document.getElementById('email').value,
                license: document.getElementById('license').value,
                vehicle: document.getElementById('vehicleNo').value,
                type: document.getElementById('vehicleType').value,
                model: document.getElementById('vehicleModel').value,
                zone: document.getElementById('zone').value,
                status: 'offline',
                trips: 0,
                rating: 5.0,
                lastSeen: 'Just now',
                lat: 23.0225 + Math.random() * 0.5,
                lng: 72.5714 + Math.random() * 0.5
            };

            if (currentDriverId) {
                const index = drivers.findIndex(d => d.id === currentDriverId);
                if (index !== -1) drivers[index] = { ...drivers[index], ...newDriver, id: currentDriverId };
                showToast('Driver updated successfully!');
            } else {
                drivers.push(newDriver);
                showToast('New driver added successfully!');
            }

            currentDriverId = null;
            closeModal('formModal');
            applyFilters();
        }
