        // ========== RENDER DRIVERS ==========
        function renderDrivers(filterStatus = 'all', searchTerm = '') {
            const grid = document.getElementById('driversGrid');
            grid.innerHTML = '';

            let filtered = drivers.filter(d => {
                const matchStatus = filterStatus === 'all' || d.status === filterStatus;
                const matchSearch = d.name.toLowerCase().includes(searchTerm.toLowerCase()) || 
                                   d.vehicle.toLowerCase().includes(searchTerm.toLowerCase()) ||
                                   d.phone.includes(searchTerm);
                return matchStatus && matchSearch;
            });

            filtered.forEach(driver => {
                const statusClass = `status-${driver.status}`;
                const statusText = driver.status === 'ontrip' ? 'On Trip' : driver.status.charAt(0).toUpperCase() + driver.status.slice(1);
                const initials = driver.name.split(' ').map(n => n[0]).join('');

                const card = document.createElement('div');
                card.className = 'driver-card';
                card.innerHTML = `

                        <div class="driver-avatar-placeholder">${initials}</div>

                            <p>${driver.phone}</p>
                        </div>

                    <div class="driver-body">

        function trackDriver(id) {

            // In real app: window.location.href = `tracking.html?driver=${id}`;
        }

        function callDriver(phone) {
            window.location.href = `tel:${phone.replace(/\s/g, '')}`;
        }
