        // ============ CALCULATE DAYS ============
        function getDaysRemaining(expiryDate) {
            if (!expiryDate) return -999;
            const today = new Date('2026-06-30');
            const expiry = new Date(expiryDate);
            const diff = expiry - today;
            return Math.ceil(diff / (1000 * 60 * 60 * 24));
        }

        function getOverallStatus(v) {
            const docs = [v.rcBook, v.insurance, v.puc, v.fitness, v.permit];
            const hasExpired = docs.some(d => d.status === 'expired');
            const hasExpiring = docs.some(d => d.status === 'expiring');
            const hasMissing = docs.some(d => d.status === 'missing');

            if (hasExpired) return 'expired';
            if (hasExpiring) return 'expiring';
            if (hasMissing) return 'missing';
            return 'valid';
        }

        function getMinDays(v) {
            const docs = [v.rcBook, v.insurance, v.puc, v.fitness, v.permit];
            let minDays = 999;
            docs.forEach(d => {
                if (d.expiry) {
                    const days = getDaysRemaining(d.expiry);
                    if (days < minDays) minDays = days;
                }
            });
            return minDays === 999 ? -999 : minDays;
        }

            let filtered = vehicles;

                filtered = filtered.filter(v => getOverallStatus(v) === currentFilter);

                const daysPercent = minDays > 30 ? 100 : minDays > 0 ? (minDays / 30) * 100 : 0;
                const daysBarClass = minDays > 30 ? 'safe' : minDays > 7 ? 'warning' : 'danger';

                return `

        function renderDocDetail(name, doc, icon) {
            const days = getDaysRemaining(doc.expiry);
            const cls = doc.status === 'missing' ? 'missing' : days < 0 ? 'missing' : days < 30 ? 'present' : 'present';
            const statusText = doc.status === 'missing' ? 'Missing' : days < 0 ? 'Expired' : days < 30 ? 'Expiring in ' + days + ' days' : 'Valid (' + days + ' days left)';
            const color = doc.status === 'missing' ? 'var(--danger)' : days < 0 ? 'var(--danger)' : days < 30 ? 'var(--warning)' : 'var(--success)';

            return `
                <div class="doc-item ${cls}">

                    <div class="doc-item-info">
                        <div class="doc-item-name">${name}</div>
                        <div class="doc-item-expiry" style="color:${color};">${statusText}</div>
                    </div>
                    ${doc.status !== 'missing' ? `<span style="font-size:12px;color:#6c757d;">${doc.expiry}</span>` : ''}
                </div>
            `;
        }
