            const vehicles = [];
            for (let i = 0; i < count; i++) {

                             status === 'idle' ? Math.floor(Math.random() * 5) : 0;
                
                vehicles.push({

                    name: vehicleNames[i % 10],

                    driver: drivers[i % 10],
                    status: status,
                    speed: speed,
                    location: `Near ${areas[i % 12]}`,

        function renderPagination(total, current) {
            const totalPages = Math.ceil(total / itemsPerPage);
            const container = document.getElementById('pagination');
            
            let html = `<button class="page-btn" ${current === 1 ? 'disabled' : ''} onclick="changePage(${current - 1})"><i class="fas fa-chevron-left"></i></button>`;
            
            for (let i = 1; i <= totalPages; i++) {
                if (i === 1 || i === totalPages || (i >= current - 1 && i <= current + 1)) {
                    html += `<button class="page-btn ${i === current ? 'active' : ''}" onclick="changePage(${i})">${i}</button>`;
                } else if (i === current - 2 || i === current + 2) {
                    html += `<button class="page-btn dots">...</button>`;
                }
            }
            
            html += `<button class="page-btn" ${current === totalPages ? 'disabled' : ''} onclick="changePage(${current + 1})"><i class="fas fa-chevron-right"></i></button>`;
            container.innerHTML = html;
        }

        function switchView(view) {
            currentView = view;

                btn.classList.toggle('active', btn.dataset.view === view);
            });
            
            document.getElementById('vehicleGrid').style.display = view === 'grid' ? 'grid' : 'none';
            document.getElementById('vehicleListView').classList.toggle('active', view === 'list');
            
            applyFilters();

        // ============ FILTERS ============
        function applyFilters() {
            const search = document.getElementById('searchInput').value.toLowerCase();
            const status = document.getElementById('statusFilter').value;
            const type = document.getElementById('typeFilter').value;
            
            let filtered = allVehicles.filter(v => {
                const matchSearch = !search || v.name.toLowerCase().includes(search) || v.number.toLowerCase().includes(search) || v.driver.toLowerCase().includes(search);
                const matchStatus = status === 'all' || v.status === status;
                const matchType = type === 'all' || v.type === type;
                return matchSearch && matchStatus && matchType;
            });
            
            if (currentView === 'grid') {
                renderGrid(filtered);
            } else {
                renderList(filtered, currentPage);
            }
            
            document.getElementById('totalVehiclesBadge').textContent = allVehicles.length;
        }
        
        document.getElementById('searchInput').addEventListener('input', () => { currentPage = 1; applyFilters(); });
        document.getElementById('statusFilter').addEventListener('change', () => { currentPage = 1; applyFilters(); });
        document.getElementById('typeFilter').addEventListener('change', () => { currentPage = 1; applyFilters(); });
        
        // ============ SELECTION ============
        function toggleSelect(id, el) {
            if (selectedVehicles.has(id)) {
                selectedVehicles.delete(id);
                el.classList.remove('checked');
                el.innerHTML = '';
            } else {
                selectedVehicles.add(id);
                el.classList.add('checked');
                el.innerHTML = '<i class="fas fa-check" style="font-size:10px;"></i>';
            }
            updateBulkActions();
        }
        
        function toggleSelectAll() {
            const el = document.getElementById('selectAll');
            const visibleVehicles = getFilteredVehicles();
            
            if (selectedVehicles.size === visibleVehicles.length) {
                selectedVehicles.clear();
                el.classList.remove('checked');
                el.innerHTML = '';
            } else {
                visibleVehicles.forEach(v => selectedVehicles.add(v.id));
                el.classList.add('checked');
                el.innerHTML = '<i class="fas fa-check" style="font-size:10px;"></i>';
            }
            
            applyFilters();
            updateBulkActions();
        }
        
        function getFilteredVehicles() {
            const search = document.getElementById('searchInput').value.toLowerCase();
            const status = document.getElementById('statusFilter').value;
            const type = document.getElementById('typeFilter').value;
            
            return allVehicles.filter(v => {
                const matchSearch = !search || v.name.toLowerCase().includes(search) || v.number.toLowerCase().includes(search);
                const matchStatus = status === 'all' || v.status === status;
                const matchType = type === 'all' || v.type === type;
                return matchSearch && matchStatus && matchType;
            });
        }
        
        function updateBulkActions() {
            const bulk = document.getElementById('bulkActions');
            const count = document.getElementById('bulkCount');
            
            if (selectedVehicles.size > 0) {
                bulk.classList.add('show');
                count.textContent = selectedVehicles.size;
            } else {
                bulk.classList.remove('show');
            }
        }
        
        function clearSelection() {
            selectedVehicles.clear();
            document.getElementById('selectAll').classList.remove('checked');
            document.getElementById('selectAll').innerHTML = '';
            applyFilters();
            updateBulkActions();
        }

            const name = document.getElementById('addName').value;
            const number = document.getElementById('addNumber').value;
            const type = document.getElementById('addType').value;
            const driver = document.getElementById('addDriver').value;
            
            if (!name || !number) {
                showToast('Please fill required fields', 'error');
                return;
            }
            
            const newVehicle = {
                id: `VEH${String(allVehicles.length + 1).padStart(5, '0')}`,
                name, number, type, driver,
                status: 'offline',
                speed: 0,
                location: 'Not tracked yet',
                lastUpdate: new Date(),
                distance: '0.0',
                fuel: parseInt(document.getElementById('addFuel').value) || 50,
                ignition: false,
                deviceId: document.getElementById('addDeviceId').value || `DEV${String(allVehicles.length + 1).padStart(6, '0')}`,
                imei: document.getElementById('addImei').value || `${Math.floor(Math.random() * 900000000000000) + 100000000000000}`
            };
            
            allVehicles.unshift(newVehicle);
            closeAddModal();
            applyFilters();
            showToast('Vehicle added successfully!', 'success');
        }
        
        // ============ TOAST ============
        function showToast(message, type = 'info') {
            const container = document.getElementById('toastContainer');
            const toast = document.createElement('div');
            toast.className = `toast ${type}`;
            toast.innerHTML = `<i class="fas fa-${type === 'success' ? 'check-circle' : type === 'error' ? 'exclamation-circle' : 'info-circle'}"></i> ${message}`;
            container.appendChild(toast);
            
            setTimeout(() => toast.classList.add('show'), 10);
            setTimeout(() => {
                toast.classList.remove('show');
                setTimeout(() => toast.remove(), 400);
            }, 3000);
        }
        
        // ============ INIT ============
        allVehicles = generateVehicles(100);
        applyFilters();
    </script>
