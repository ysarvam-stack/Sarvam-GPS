    <script>
        // ============ AUTH ============
        const token = localStorage.getItem('sarvam_token');
        if (!token) window.location.href = 'login.html';

        const user = JSON.parse(localStorage.getItem('sarvam_user') || '{}');
        document.getElementById('userName').textContent = user.name || 'Admin';
        document.getElementById('userAvatar').textContent = (user.name || 'A').charAt(0).toUpperCase();

        // ============ SIDEBAR ============
        const sidebar = document.getElementById('sidebar');
        document.getElementById('toggleSidebar').addEventListener('click', () => {
            sidebar.classList.toggle('collapsed');
        });

        document.getElementById('logoutBtn').addEventListener('click', () => {
            localStorage.removeItem('sarvam_token');
            localStorage.removeItem('sarvam_user');
            window.location.href = 'login.html';
        });

        // ============ REPORT DATA ============
        let currentReport = 'distance';

        const monthlyData = {
            'Toyota Innova': [1250, 1380, 1420, 1350, 1480, 1520],
            'Tata Ace': [980, 1100, 1050, 1200, 1150, 1300],
            'Mahindra Bolero': [1500, 1450, 1600, 1550, 1620, 1700],
            'Maruti Swift': [800, 850, 900, 880, 920, 950],
            'Honda Activa': [450, 500, 480, 520, 550, 580],
            'Force Traveller': [1800, 1750, 1900, 1850, 1920, 2000]
        };

        const months = ['Jan', 'Feb', 'Mar', 'Apr', 'May', 'Jun'];

        // ============ SELECT REPORT ============
        function selectReport(type) {
            currentReport = type;
            document.querySelectorAll('.report-cat-card').forEach(card => {
                card.classList.remove('active');
            });
            event.currentTarget.classList.add('active');
            generateCharts();
        }

        // ============ GENERATE CHARTS ============
        function generateCharts() {
            // Distance Chart
            const distanceChart = document.getElementById('distanceChart');
            const colors = ['#3498db', '#e94560', '#2ecc71', '#f39c12', '#9b59b6', '#1abc9c'];
            const maxVal = 2000;

            distanceChart.innerHTML = months.map((m, i) => {
                const val = monthlyData['Toyota Innova'][i];
                const height = (val / maxVal) * 100;
                return `
                    <div class="bar-item">
                        <div class="bar" style="height:${height}%;background:${colors[i]};">
                            <span class="bar-value">${val}</span>
                        </div>
                        <span class="bar-label">${m}</span>
                    </div>
                `;
            }).join('');

            // Utilization Chart
            const utilChart = document.getElementById('utilizationChart');
            const vehicles = Object.keys(monthlyData);
            const utilColors = ['#e94560', '#3498db', '#2ecc71', '#f39c12', '#9b59b6', '#1abc9c'];

            utilChart.innerHTML = vehicles.map((v, i) => {
                const total = monthlyData[v].reduce((a, b) => a + b, 0);
                const height = (total / 12000) * 100;
                return `
                    <div class="bar-item">
                        <div class="bar" style="height:${height}%;background:${utilColors[i]};">
                            <span class="bar-value">${total}</span>
                        </div>
                        <span class="bar-label" style="font-size:10px;">${v.split(' ')[0]}</span>
                    </div>
                `;
            }).join('');
        }

        // ============ RENDER KM TABLE ============
        function renderKmTable() {
            const tbody = document.getElementById('kmTableBody');

            tbody.innerHTML = Object.entries(monthlyData).map(([vehicle, data]) => {
                const total = data.reduce((a, b) => a + b, 0);
                const avg = Math.round(total / 6);
                const maxVal = Math.max(...data);

                return `
                    <tr>
                        <td>
                            <div style="font-weight:600;">${vehicle}</div>
                        </td>
                        ${data.map((km, i) => `
                            <td>
                                ${km}
                                ${i === 5 ? '<span class="month-badge current">Current</span>' : ''}
                            </td>
                        `).join('')}
                        <td><strong>${total.toLocaleString()}</strong></td>
                        <td>
                            ${avg}
                            <div class="km-bar">
                                <div class="km-bar-fill" style="width:${(avg/2000)*100}%;background:${avg > 1500 ? 'var(--success)' : avg > 1000 ? 'var(--warning)' : 'var(--danger)'};"></div>
                            </div>
                        </td>
                    </tr>
                `;
            }).join('');
        }

        // ============ GENERATE REPORT ============
        function generateReport() {
            showToast('Report generated successfully!', 'success');
            generateCharts();
            renderKmTable();
        }

        // ============ EXPORT ============
        function exportReport(format) {
            const formats = {
                pdf: 'PDF',
                excel: 'Excel',
                csv: 'CSV',
                print: 'Print'
            };
            showToast(`Report exported as ${formats[format]}!`, 'success');
        }

        // ============ TOAST ============
        function showToast(message, type) {
            const container = document.getElementById('toastContainer');
            const toast = document.createElement('div');
            toast.className = `toast ${type}`;
            toast.innerHTML = `<i class="fas fa-${type === 'success' ? 'check-circle' : 'info-circle'}"></i> ${message}`;
            container.appendChild(toast);
            setTimeout(() => toast.classList.add('show'), 10);
            setTimeout(() => {
                toast.classList.remove('show');
                setTimeout(() => toast.remove(), 400);
            }, 3000);
        }

        // ============ INIT ============
        generateCharts();
        renderKmTable();
