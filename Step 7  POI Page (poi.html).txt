            document.getElementById('popupLat').textContent = p.lat.toFixed(4);
            document.getElementById('popupLng').textContent = p.lng.toFixed(4);
            document.getElementById('popupAddress').textContent = p.address;

            // Center map
            map.setView([p.lat, p.lng], 15);
            if (markers[id]) markers[id].openPopup();
        }

        function savePoi() {
            const name = document.getElementById('poiName').value;
            const type = document.getElementById('poiType').value;
            const lat = parseFloat(document.getElementById('poiLat').value);
            const lng = parseFloat(document.getElementById('poiLng').value);
            const radius = parseInt(document.getElementById('poiRadius').value) || 100;
            const address = document.getElementById('poiAddress').value;

            if (!name || isNaN(lat) || isNaN(lng)) {
                showToast('Please fill all required fields', 'error');
                return;
            }

            const newPoi = {
                id: 'POI' + String(pois.length + 1).padStart(3, '0'),
                name,
                type,
                lat,
                lng,
                address: address || 'Address not specified',
                radius,
                color: selectedColor,
                distance: (Math.random() * 10).toFixed(1)
            };

            pois.push(newPoi);
            closeAddModal();
            renderPoiList();
            renderMarkers();
            showToast('POI added successfully!', 'success');
        }
