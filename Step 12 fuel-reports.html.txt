                data1 = [45, 52, 38, 65, 48, 70, 55];
                data2 = [360, 416, 304, 520, 384, 560, 440];

        // ========== EFFICIENCY CALCULATOR ==========
        function calculateEfficiency() {
            const distance = parseFloat(document.getElementById('calcDistance').value) || 0;
            const fuel = parseFloat(document.getElementById('calcFuel').value) || 0;
            const price = parseFloat(document.getElementById('calcPrice').value) || 0;

            if (distance > 0 && fuel > 0) {
                const efficiency = (distance / fuel).toFixed(2);
                const totalCost = (fuel * price).toFixed(0);
                const costPerKm = (totalCost / distance).toFixed(2);

                document.getElementById('efficiencyValue').textContent = efficiency;
                document.getElementById('totalCost').textContent = totalCost;
                document.getElementById('costPerKm').textContent = costPerKm;
                document.getElementById('calcResult').style.display = 'block';
            }
        }
