                    <div class="progress-section">
                        <div class="progress-label">
                            <span style="font-size: 12px; color: #666;">Battery Level</span>
                            <span style="font-size: 12px; font-weight: 600; color: ${device.battery > 50 ? 'var(--success)' : device.battery > 20 ? 'var(--warning)' : 'var(--danger)'};">${device.battery}%</span>
                        </div>
                        <div class="progress-bar">
                            <div class="progress-fill ${batteryColor}" style="width: ${device.battery}%;"></div>
                        </div>
                    </div>
                    <div class="device-actions">
                        <button class="btn btn-primary" onclick="rebootDevice('${device.id}')"><i class="fas fa-sync"></i> Reboot</button>
                        <button class="btn btn-outline" onclick="updateFirmware('${device.id}')"><i class="fas fa-download"></i> Update</button>
                        <button class="btn btn-outline" onclick="deviceDiagnostics('${device.id}')"><i class="fas fa-stethoscope"></i> Diagnose</button>
                    </div>
                `;
                grid.appendChild(card);
            });
        }

        // ========== FILTER ==========
        function filterDevices() {
            const type = document.getElementById('typeFilter').value;
            const status = document.getElementById('statusFilter').value;
            const vehicle = document.getElementById('vehicleFilter').value;

            const filtered = devices.filter(d => {
                const matchType = type === 'all' || d.type === type;
                const matchStatus = status === 'all' || d.status === status;
                const matchVehicle = vehicle === 'all' || d.vehicle === vehicle;
                return matchType && matchStatus && matchVehicle;
            });

            renderDevices(filtered);
        }

        document.getElementById('searchInput').addEventListener('input', (e) => {
            const term = e.target.value.toLowerCase();
            const filtered = devices.filter(d => 
                d.name.toLowerCase().includes(term) || 
                d.id.toLowerCase().includes(term) ||
                d.vehicle.toLowerCase().includes(term) ||
                d.imei.includes(term)
            );
            renderDevices(filtered);
        });

        // ========== ACTIONS ==========
        function rebootDevice(id) {
            showToast(`Rebooting device ${id}...`);
            setTimeout(() => showToast(`Device ${id} rebooted successfully`), 2000);
        }

        function updateFirmware(id) {
            showToast(`Updating firmware for ${id}...`);
            setTimeout(() => showToast(`Firmware updated for ${id}`), 3000);
        }

        function deviceDiagnostics(id) {
            showToast(`Running diagnostics on ${id}...`);
        }

        function addDevice() {
            showToast('Add Device modal opened');
        }

        function saveConfig() {
            const tank = document.getElementById('tankCapacity').value;
            const smoothing = document.getElementById('smoothingFactor').value;
            showToast(`Config saved: Tank ${tank}L, Smoothing ${smoothing}%`);
        }

        // ========== TOAST ==========
        function showToast(message) {
            const toast = document.getElementById('toast');
            document.getElementById('toastMessage').textContent = message;
            toast.style.display = 'flex';
            setTimeout(() => toast.style.display = 'none', 3000);
        }

        renderDevices();
