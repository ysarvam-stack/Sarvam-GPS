            map = L.map('map', { zoomControl: false }).setView([23.0225, 72.5714], 12);

            L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png', {
                attribution: 'Sarvam GPS',
                maxZoom: 19
            }).addTo(map);

            drawnItems = L.featureGroup().addTo(map);

            // Click on map to create geofence
            map.on('click', function(e) {
                if (drawMode === 'circle') {
                    createCircleGeofence(e.latlng);
                }
            });
        }

        // ============ RENDER LIST ============
        function renderGeofenceList() {
            const list = document.getElementById('geofenceList');
            const search = document.getElementById('searchGeofence').value.toLowerCase();

            const filtered = geofences.filter(g => g.name.toLowerCase().includes(search));

            list.innerHTML = filtered.map(g => `
                <div class="geofence-item ${selectedGeofenceId === g.id ? 'active' : ''}" onclick="selectGeofence('${g.id}')">
                    <div class="geofence-icon ${g.type}">
                        <i class="fas fa-${g.type === 'circle' ? 'circle' : g.type === 'polygon' ? 'draw-polygon' : 'square'}"></i>
                    </div>
                    <div class="geofence-info">
                        <div class="geofence-name">${g.name}</div>
                        <div class="geofence-meta">
                            <span><i class="fas fa-ruler"></i> ${g.radius}m</span>
                            <span><i class="fas fa-map-marker-alt"></i> ${g.lat.toFixed(3)}, ${g.lng.toFixed(3)}</span>
                        </div>
                    </div>
                    <div class="geofence-status ${g.status}"></div>
                </div>
            `).join('');
        }

        // ============ RENDER ON MAP ============
        function renderGeofencesOnMap() {
            drawnItems.clearLayers();

            geofences.forEach(g => {
                let layer;
                const color = g.status === 'active' ? '#e94560' : '#95a5a6';

                if (g.type === 'circle') {
                    layer = L.circle([g.lat, g.lng], {
                        radius: g.radius,
                        color: color,
                        fillColor: color,
                        fillOpacity: 0.15,
                        weight: 2
                    });
                } else if (g.type === 'rectangle') {
                    const bounds = [
                        [g.lat - 0.005, g.lng - 0.005],
                        [g.lat + 0.005, g.lng + 0.005]
                    ];
                    layer = L.rectangle(bounds, {
                        color: color,
                        fillColor: color,
                        fillOpacity: 0.15,
                        weight: 2
                    });
                } else {
                    const points = [
                        [g.lat, g.lng],
                        [g.lat + 0.008, g.lng + 0.003],
                        [g.lat + 0.003, g.lng + 0.01],
                        [g.lat - 0.005, g.lng + 0.005]
                    ];
                    layer = L.polygon(points, {
                        color: color,
                        fillColor: color,
                        fillOpacity: 0.15,
                        weight: 2
                    });
                }

                layer.geofenceId = g.id;
                layer.on('click', () => selectGeofence(g.id));
                drawnItems.addLayer(layer);
            });
        }

        // ============ SELECT GEOFENCE ============
        function selectGeofence(id) {
            selectedGeofenceId = id;
            const g = geofences.find(ge => ge.id === id);
            if (!g) return;

            renderGeofenceList();

            // Show detail panel
            document.getElementById('geofenceDetail').classList.add('show');
            document.getElementById('gfName').value = g.name;
            document.getElementById('gfType').value = g.type;
            document.getElementById('gfRadius').value = g.radius;
            document.getElementById('gfDesc').value = g.description || '';

            // Center map
            map.setView([g.lat, g.lng], 14);
        }

        // ============ DRAW MODE ============
        function setDrawMode(mode) {
            drawMode = mode;
            document.querySelectorAll('.draw-btn').forEach(btn => btn.classList.remove('active'));
            document.getElementById('btn' + mode.charAt(0).toUpperCase() + mode.slice(1)).classList.add('active');

            if (mode === 'circle') {
                map.getContainer().style.cursor = 'crosshair';
                showToast('Click on map to place circle geofence', 'info');
            } else if (mode === 'polygon') {
                showToast('Polygon drawing: Click multiple points, double-click to finish', 'info');
                startPolygonDraw();
            } else if (mode === 'rectangle') {
                showToast('Rectangle drawing: Click and drag on map', 'info');
                startRectangleDraw();
            }
        }

        function createCircleGeofence(latlng) {
            const id = 'GF' + String(geofences.length + 1).padStart(3, '0');
            const newGeofence = {
                id: id,
                name: 'New Geofence ' + (geofences.length + 1),
                type: 'circle',
                lat: latlng.lat,
                lng: latlng.lng,
                radius: 500,
                description: '',
                status: 'active',
                alerts: { entry: true, exit: true, speed: false }
            };

            geofences.push(newGeofence);
            renderGeofencesOnMap();
            selectGeofence(id);
            drawMode = null;
            map.getContainer().style.cursor = '';
            document.querySelectorAll('.draw-btn').forEach(btn => btn.classList.remove('active'));
            showToast('Circle geofence created! Edit details below.', 'success');
        }

        let polygonPoints = [];
        let tempPolygon = null;

        function startPolygonDraw() {
            polygonPoints = [];
            if (tempPolygon) map.removeLayer(tempPolygon);

            map.on('click', onPolygonClick);
            map.on('dblclick', onPolygonFinish);
        }

        function onPolygonClick(e) {
            polygonPoints.push([e.latlng.lat, e.latlng.lng]);

            if (tempPolygon) map.removeLayer(tempPolygon);
            if (polygonPoints.length > 2) {
                tempPolygon = L.polygon(polygonPoints, {
                    color: '#e94560',
                    fillColor: '#e94560',
                    fillOpacity: 0.2,
                    weight: 2,
                    dashArray: '5, 5'
                }).addTo(map);
            }

            // Add point marker
            L.circleMarker(e.latlng, {
                radius: 4,
                color: '#e94560',
                fillColor: '#e94560',
                fillOpacity: 1
            }).addTo(map);
        }

        function onPolygonFinish(e) {
            map.off('click', onPolygonClick);
            map.off('dblclick', onPolygonFinish);

            if (polygonPoints.length < 3) {
                showToast('Need at least 3 points for polygon', 'error');
                return;
            }

            const center = polygonPoints.reduce((acc, p) => [acc[0] + p[0], acc[1] + p[1]], [0, 0]);
            center[0] /= polygonPoints.length;
            center[1] /= polygonPoints.length;

            const id = 'GF' + String(geofences.length + 1).padStart(3, '0');
            const newGeofence = {
                id: id,
                name: 'Polygon ' + (geofences.length + 1),
                type: 'polygon',
                lat: center[0],
                lng: center[1],
                radius: 300,
                description: '',
                status: 'active',
                alerts: { entry: true, exit: true, speed: false }
            };

            geofences.push(newGeofence);
            renderGeofencesOnMap();
            selectGeofence(id);
            drawMode = null;
            document.querySelectorAll('.draw-btn').forEach(btn => btn.classList.remove('active'));
            showToast('Polygon geofence created!', 'success');
        }

        function startRectangleDraw() {
            let startPoint = null;
            let tempRect = null;

            map.on('mousedown', function(e) {
                startPoint = e.latlng;
            });

            map.on('mousemove', function(e) {
                if (!startPoint) return;
                if (tempRect) map.removeLayer(tempRect);

                const bounds = [
                    [startPoint.lat, startPoint.lng],
                    [e.latlng.lat, e.latlng.lng]
                ];
                tempRect = L.rectangle(bounds, {
                    color: '#e94560',
                    fillColor: '#e94560',
                    fillOpacity: 0.2,
                    weight: 2
                }).addTo(map);
            });

            map.on('mouseup', function(e) {
                if (!startPoint) return;

                const bounds = [
                    [startPoint.lat, startPoint.lng],
                    [e.latlng.lat, e.latlng.lng]
                ];

                const center = [
                    (startPoint.lat + e.latlng.lat) / 2,
                    (startPoint.lng + e.latlng.lng) / 2
                ];

                const id = 'GF' + String(geofences.length + 1).padStart(3, '0');
                const newGeofence = {
                    id: id,
                    name: 'Rectangle ' + (geofences.length + 1),
                    type: 'rectangle',
                    lat: center[0],
                    lng: center[1],
                    radius: 600,
                    description: '',
                    status: 'active',
                    alerts: { entry: true, exit: true, speed: false }
                };

                geofences.push(newGeofence);
                renderGeofencesOnMap();
                selectGeofence(id);
                drawMode = null;
                startPoint = null;
                document.querySelectorAll('.draw-btn').forEach(btn => btn.classList.remove('active'));
                showToast('Rectangle geofence created!', 'success');
            });
        }

        // ============ SAVE / DELETE ============
        function saveGeofence() {
            if (!selectedGeofenceId) return;

            const g = geofences.find(ge => ge.id === selectedGeofenceId);
            if (g) {
                g.name = document.getElementById('gfName').value || g.name;
                g.type = document.getElementById('gfType').value;
                g.radius = parseInt(document.getElementById('gfRadius').value) || g.radius;
                g.description = document.getElementById('gfDesc').value;

                renderGeofenceList();
                renderGeofencesOnMap();
                showToast('Geofence saved successfully!', 'success');
            }
        }

        function deleteGeofence() {
            if (!selectedGeofenceId) return;

            if (confirm('Are you sure you want to delete this geofence?')) {
                geofences = geofences.filter(g => g.id !== selectedGeofenceId);
                selectedGeofenceId = null;
                closeDetail();
                renderGeofenceList();
                renderGeofencesOnMap();
                showToast('Geofence deleted!', 'success');
            }
        }

        function closeDetail() {
            document.getElementById('geofenceDetail').classList.remove('show');
            selectedGeofenceId = null;
            renderGeofenceList();
        }

        function fitAllGeofences() {
            if (drawnItems.getLayers().length > 0) {
                map.fitBounds(drawnItems.getBounds().pad(0.1));
            }
        }

        // ============ SEARCH ============
        document.getElementById('searchGeofence').addEventListener('input', renderGeofenceList);

        // ============ TOAST ============
        function showToast(message, type) {
            const container = document.getElementById('toastContainer');
            const toast = document.createElement('div');
            toast.className = `toast ${type}`;
            toast.innerHTML = `<i class="fas fa-${type === 'success' ? 'check-circle' : type === 'error' ? 'exclamation-circle' : 'info-circle'}"></i> ${message}`;
            container.appendChild(toast);
            setTimeout(() => toast.classList.add('show'), 10);
            setTimeout(() => {
                toast.classList.remove('show');
                setTimeout(() => toast.remove(), 400);
            }, 3000);
        }

        // ============ INIT ============
        initMap();
        loadDemoGeofences();
    </script>
