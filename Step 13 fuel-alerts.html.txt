        // ========== RENDER ALERTS ==========
        function renderAlerts(data = alerts) {
            const container = document.getElementById('alertsContainer');
            container.innerHTML = '';

            data.forEach(alert => {
                const card = document.createElement('div');
                card.className = `alert-card ${alert.type}`;
                
                let iconClass, badgeClass, badgeText;
                switch(alert.type) {
                    case 'theft': iconClass = 'theft'; badgeClass = 'badge-danger'; badgeText = 'THEFT'; break;
                    case 'low': iconClass = 'low'; badgeClass = 'badge-warning'; badgeText = 'LOW FUEL'; break;
                    case 'sudden': iconClass = 'sudden'; badgeClass = 'badge-info'; badgeText = 'SUDDEN DROP'; break;
                    case 'leak': iconClass = 'leak'; badgeClass = 'badge-purple'; badgeText = 'LEAK'; break;
                }

                const iconMap = { theft: 'fa-user-secret', low: 'fa-gas-pump', sudden: 'fa-chart-line', leak: 'fa-tint' };

                card.innerHTML = `
                    <div class="alert-icon ${iconClass}"><i class="fas ${iconMap[alert.type]}"></i></div>
                    <div class="alert-content">
                        <div style="display: flex; justify-content: space-between; align-items: center; margin-bottom: 5px;">
                            <h4>${alert.title}</h4>
                            <span class="badge ${badgeClass}">${badgeText}</span>
                        </div>
                        <p>${alert.message}</p>
                        <div class="alert-meta">
                            <span><i class="fas fa-car"></i> ${alert.vehicle}</span>
                            <span><i class="fas fa-user"></i> ${alert.driver}</span>
                            <span><i class="fas fa-clock"></i> ${alert.time}</span>
                            <span><i class="fas fa-map-marker-alt"></i> ${alert.location}</span>
                            <span><i class="fas fa-tint"></i> Drop: ${alert.fuelDrop}</span>
                        </div>
                    </div>
                    <div class="alert-actions">
                        ${alert.status === 'active' ? 
                            `<button class="btn btn-success" onclick="resolveAlert(${alert.id})"><i class="fas fa-check"></i> Resolve</button>
                             <button class="btn btn-outline" onclick="viewDetails(${alert.id})"><i class="fas fa-eye"></i></button>` :
                            `<button class="btn btn-outline" disabled><i class="fas fa-check-circle"></i> Resolved</button>`
                        }
                    </div>
                `;
                container.appendChild(card);
            });
        }

            const type = document.getElementById('typeFilter').value;
            const status = document.getElementById('statusFilter').value;
            const vehicle = document.getElementById('vehicleFilter').value;

            const filtered = alerts.filter(a => {
                const matchType = type === 'all' || a.type === type;
                const matchStatus = status === 'all' || a.status === status;
                const matchVehicle = vehicle === 'all' || a.vehicle === vehicle;
                return matchType && matchStatus && matchVehicle;
            });

            renderAlerts(filtered);
        }

        document.getElementById('searchInput').addEventListener('input', (e) => {
            const term = e.target.value.toLowerCase();
            const filtered = alerts.filter(a => 
                a.title.toLowerCase().includes(term) || 
                a.vehicle.toLowerCase().includes(term) ||
                a.driver.toLowerCase().includes(term)
            );
            renderAlerts(filtered);
        });

        // ========== ACTIONS ==========
        function resolveAlert(id) {
            const alert = alerts.find(a => a.id === id);
            if (alert) {
                alert.status = 'resolved';
                alert.time = 'Just now';
                renderAlerts();
                showToast(`Alert resolved: ${alert.title}`);
            }
        }

        function viewDetails(id) {
            const alert = alerts.find(a => a.id === id);
            showToast(`Viewing details for ${alert.vehicle}`);
        }

        function markAllRead() {
            alerts.forEach(a => a.status = 'resolved');
            renderAlerts();
            showToast('All alerts marked as resolved');
        }

        function saveSettings() {
            const lowFuel = document.getElementById('lowFuelThreshold').value;
            const suddenDrop = document.getElementById('suddenDropThreshold').value;
            const theft = document.getElementById('theftThreshold').value;
            showToast(`Settings saved: Low Fuel ${lowFuel}%, Sudden Drop ${suddenDrop}L, Theft ${theft}L`);
        }

        // ========== TOAST ==========
        function showToast(message) {
            const toast = document.getElementById('toast');
            document.getElementById('toastMessage').textContent = message;
            toast.style.display = 'flex';
            setTimeout(() => toast.style.display = 'none', 3000);
        }

        // ========== INIT ==========
        renderAlerts();
    </script>
