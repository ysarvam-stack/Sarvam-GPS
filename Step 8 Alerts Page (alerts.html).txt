        // ============ RENDER ALERTS ============
        function renderAlerts() {
            const list = document.getElementById('alertList');
            const search = document.getElementById('searchAlert').value.toLowerCase();

            let filtered = alerts;
            if (currentFilter === 'unread') {
                filtered = filtered.filter(a => !a.read);
            } else if (currentFilter !== 'all') {
                filtered = filtered.filter(a => a.type === currentFilter);
            }

            if (search) {
                filtered = filtered.filter(a => 
                    a.title.toLowerCase().includes(search) || 
                    a.vehicle.toLowerCase().includes(search) ||
                    a.description.toLowerCase().includes(search)
                );
            }

            if (filtered.length === 0) {
                list.innerHTML = `
                    <div style="text-align:center;padding:60px;color:#6c757d;">
                        <i class="fas fa-bell-slash" style="font-size:48px;opacity:0.3;margin-bottom:16px;display:block;"></i>
                        <p>No alerts found</p>
                    </div>
                `;
                return;
            }

            list.innerHTML = filtered.map(a => `
                <div class="alert-card ${a.read ? 'read' : 'unread'}" onclick="viewAlert(${a.id})">
                    <div class="alert-icon ${a.type}">
                        <i class="fas fa-${alertIcons[a.type] || 'bell'}"></i>
                    </div>
                    <div class="alert-content">
                        <div class="alert-title">
                            ${a.title}
                            <span class="alert-badge ${a.severity}">${a.severity}</span>
                            ${!a.read ? '<span style="width:8px;height:8px;background:var(--primary);border-radius:50%;"></span>' : ''}
                        </div>
                        <div class="alert-description">${a.description}</div>
                        <div class="alert-meta">
                            <span><i class="fas fa-car"></i> ${a.vehicle}</span>
                            <span><i class="fas fa-user"></i> ${a.driver}</span>
                            <span><i class="fas fa-map-marker-alt"></i> ${a.location}</span>
                        </div>
                    </div>
                    <div class="alert-time">${a.time}</div>
                    <div class="alert-actions" onclick="event.stopPropagation()">
                        <button class="alert-action-btn" title="Mark as read" onclick="markRead(${a.id})">
                            <i class="fas fa-check"></i>
                        </button>
                        <button class="alert-action-btn" title="View on map" onclick="viewOnMap(${a.id})">
                            <i class="fas fa-map-marker-alt"></i>
                        </button>
                    </div>
                </div>
            `).join('');
        }

        // ============ FILTER ============
        function filterAlerts(filter) {
            currentFilter = filter;
            document.querySelectorAll('.filter-chip').forEach(btn => {
                btn.classList.toggle('active', btn.dataset.filter === filter);
            });
            renderAlerts();
        }

        document.getElementById('searchAlert').addEventListener('input', renderAlerts);

        // ============ ACTIONS ============
        function viewAlert(id) {
            const a = alerts.find(al => al.id === id);
            if (a) {
                a.read = true;
                renderAlerts();
                updateCounts();
            }
        }

        function markRead(id) {
            const a = alerts.find(al => al.id === id);
            if (a) {
                a.read = true;
                renderAlerts();
                updateCounts();
                showToast('Marked as read', 'success');
            }
        }

        function markAllRead() {
            alerts.forEach(a => a.read = true);
            renderAlerts();
            updateCounts();
            showToast('All alerts marked as read', 'success');
        }

        function viewOnMap(id) {
            const a = alerts.find(al => al.id === id);
            if (a) {
                window.location.href = `live-tracking.html?vehicle=${a.vehicle.match(/GJ-\d{2}-[A-Z]{2}-\d{4}/)?.[0]}`;
            }
        }

        // ============ UPDATE COUNTS ============
        function updateCounts() {
            const unread = alerts.filter(a => !a.read).length;
            const speed = alerts.filter(a => a.type === 'speed').length;
            const geofence = alerts.filter(a => a.type === 'geofence').length;
            const idle = alerts.filter(a => a.type === 'idle').length;
            const ignition = alerts.filter(a => a.type === 'ignition').length;
            const sos = alerts.filter(a => a.type === 'sos').length;

            document.getElementById('alertCount').textContent = unread;
            document.getElementById('countAll').textContent = alerts.length;
            document.getElementById('countUnread').textContent = unread;
            document.getElementById('countSpeed').textContent = speed;
            document.getElementById('countGeofence').textContent = geofence;
            document.getElementById('countSos').textContent = sos;

            document.getElementById('speedCount').textContent = speed;
            document.getElementById('geofenceCount').textContent = geofence;
            document.getElementById('idleCount').textContent = idle;
            document.getElementById('ignitionCount').textContent = ignition;
        }

        // ============ TOAST ============
        function showToast(message, type) {
            const container = document.getElementById('toastContainer');
            const toast = document.createElement('div');
            toast.className = `toast ${type}`;
            toast.innerHTML = `<i class="fas fa-${type === 'success' ? 'check-circle' : 'info-circle'}"></i> ${message}`;
            container.appendChild(toast);
            setTimeout(() => toast.classList.add('show'), 10);
            setTimeout(() => {
                toast.classList.remove('show');
                setTimeout(() => toast.remove(), 400);
            }, 3000);
        }

        // ============ INIT ============
        renderAlerts();
        updateCounts();
