        const token = localStorage.getItem('sarvam_token');
        if (!token) window.location.href = 'login.html';

        const user = JSON.parse(localStorage.getItem('sarvam_user') || '{}');
        document.getElementById('userName').textContent = user.name || 'Admin';
        document.getElementById('userAvatar').textContent = (user.name || 'A').charAt(0).toUpperCase();

        // ============ SIDEBAR ============
        const sidebar = document.getElementById('sidebar');
        document.getElementById('toggleSidebar').addEventListener('click', () => {
            sidebar.classList.toggle('collapsed');
        });

        document.getElementById('logoutBtn').addEventListener('click', () => {
            localStorage.removeItem('sarvam_token');
            localStorage.removeItem('sarvam_user');
            window.location.href = 'login.html';
        });

        // ============ MAP ============
        let map, routePolyline, movingMarker, startMarker, endMarker;
        let routePoints = [];
        let isPlaying = false;
        let playbackInterval;
        let currentPoint = 0;
        let playbackSpeed = 1;
        let satelliteMode = false;

        function initMap() {
            map = L.map('map', { zoomControl: false }).setView([23.0225, 72.5714], 12);

            L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png', {
                attribution: 'Sarvam GPS',
                maxZoom: 19
            }).addTo(map);
        }

        // ============ DEMO DATA ============
        const vehicles = [
            { id: 'VEH00001', name: 'Toyota Innova', number: 'GJ-01-AA-0001' },
            { id: 'VEH00002', name: 'Tata Ace', number: 'GJ-02-BB-0002' },
            { id: 'VEH00003', name: 'Mahindra Bolero', number: 'GJ-03-CC-0003' },
            { id: 'VEH00004', name: 'Maruti Swift', number: 'GJ-04-DD-0004' },
            { id: 'VEH00005', name: 'Honda Activa', number: 'GJ-05-EE-0005' }
        ];

        // Populate vehicle select
        const select = document.getElementById('vehicleSelect');
        vehicles.forEach(v => {
            const option = document.createElement('option');
            option.value = v.id;
            option.textContent = `${v.name} (${v.number})`;
            select.appendChild(option);
        });

        // URL param
        const urlParams = new URLSearchParams(window.location.search);
        const preselectVehicle = urlParams.get('vehicle');
        if (preselectVehicle) {
            select.value = preselectVehicle;
            loadRoute();
        }

        // ============ GENERATE ROUTE ============
        function generateRoutePoints() {
            const points = [];
            const baseLat = 23.0225;
            const baseLng = 72.5714;
            const numPoints = 50;

            for (let i = 0; i < numPoints; i++) {
                const angle = (i / numPoints) * Math.PI * 4;
                const radius = 0.15 * (i / numPoints);
                points.push({
                    lat: baseLat + Math.cos(angle) * radius + (Math.random() - 0.5) * 0.02,
                    lng: baseLng + Math.sin(angle) * radius + (Math.random() - 0.5) * 0.02,
                    speed: Math.floor(Math.random() * 60) + 10,
                    time: new Date(2026, 5, 29, 8, i * 3, 0),
                    status: i % 10 === 0 ? 'stop' : 'moving'
                });
            }
            return points;
        }

        function generateStops() {
            return [
                { location: 'SG Highway, Ahmedabad', duration: '15 min', time: '08:30 AM', lat: 23.035, lng: 72.510 },
                { location: 'Ashram Road, Ahmedabad', duration: '25 min', time: '09:45 AM', lat: 23.030, lng: 72.580 },
                { location: 'Maninagar, Ahmedabad', duration: '10 min', time: '11:20 AM', lat: 23.000, lng: 72.600 },
                { location: 'Bopal, Ahmedabad', duration: '30 min', time: '01:15 PM', lat: 23.010, lng: 72.480 }
            ];
        }

        // ============ LOAD ROUTE ============
        function loadRoute() {
            const vehicleId = document.getElementById('vehicleSelect').value;
            if (!vehicleId) return;

            routePoints = generateRoutePoints();
            const stops = generateStops();

            // Clear previous
            if (routePolyline) map.removeLayer(routePolyline);
            if (movingMarker) map.removeLayer(movingMarker);
            if (startMarker) map.removeLayer(startMarker);
            if (endMarker) map.removeLayer(endMarker);

            // Draw route
            const latlngs = routePoints.map(p => [p.lat, p.lng]);
            routePolyline = L.polyline(latlngs, {
                color: '#e94560',
                weight: 4,
                opacity: 0.8,
                lineCap: 'round',
                lineJoin: 'round'
            }).addTo(map);

            // Start and end markers
            const startIcon = L.divIcon({
                className: 'custom-marker',
                html: '<div style="background:#2ecc71;width:24px;height:24px;border-radius:50%;display:flex;align-items:center;justify-content:center;color:white;font-size:12px;border:3px solid white;box-shadow:0 2px 8px rgba(46,204,113,0.4);"><i class="fas fa-play"></i></div>',
                iconSize: [24, 24],
                iconAnchor: [12, 12]
            });

            const endIcon = L.divIcon({
                className: 'custom-marker',
                html: '<div style="background:#e74c3c;width:24px;height:24px;border-radius:50%;display:flex;align-items:center;justify-content:center;color:white;font-size:12px;border:3px solid white;box-shadow:0 2px 8px rgba(231,76,60,0.4);"><i class="fas fa-flag"></i></div>',
                iconSize: [24, 24],
                iconAnchor: [12, 12]
            });

            startMarker = L.marker([routePoints[0].lat, routePoints[0].lng], { icon: startIcon }).addTo(map);
            endMarker = L.marker([routePoints[routePoints.length - 1].lat, routePoints[routePoints.length - 1].lng], { icon: endIcon }).addTo(map);

            // Moving marker
            const movingIcon = L.divIcon({
                className: 'custom-marker',
                html: '<div style="background:#e94560;width:32px;height:32px;border-radius:50%;display:flex;align-items:center;justify-content:center;color:white;font-size:14px;border:3px solid white;box-shadow:0 3px 12px rgba(233,69,96,0.4);"><i class="fas fa-car"></i></div>',
                iconSize: [32, 32],
                iconAnchor: [16, 16]
            });

            movingMarker = L.marker([routePoints[0].lat, routePoints[0].lng], { icon: movingIcon }).addTo(map);

            // Fit bounds
            map.fitBounds(routePolyline.getBounds().pad(0.1));

            // Show summary
            const totalDist = calculateDistance(routePoints);
            const totalTime = routePoints[routePoints.length - 1].time - routePoints[0].time;
            const hours = Math.floor(totalTime / 3600000);
            const minutes = Math.floor((totalTime % 3600000) / 60000);
            const speeds = routePoints.map(p => p.speed);
            const avgSpeed = Math.round(speeds.reduce((a, b) => a + b, 0) / speeds.length);
            const maxSpeed = Math.max(...speeds);

            document.getElementById('routeSummary').style.display = 'block';
            document.getElementById('totalDistance').textContent = totalDist.toFixed(1) + ' km';
            document.getElementById('totalTime').textContent = `${hours}h ${minutes}m`;
            document.getElementById('avgSpeed').textContent = avgSpeed + ' km/h';
            document.getElementById('maxSpeed').textContent = maxSpeed + ' km/h';
            document.getElementById('stopCount').textContent = stops.length;

            // Show timeline
            renderTimeline(routePoints);

            // Show stops
            document.getElementById('stopsSection').style.display = 'block';
            document.getElementById('stopsList').innerHTML = stops.map((s, i) => `
                <div class="stop-item" onclick="focusStop(${s.lat}, ${s.lng})">
                    <div class="stop-icon"><i class="fas fa-hand-paper"></i></div>
                    <div class="stop-info">
                        <div class="stop-location">${s.location}</div>
                        <div class="stop-duration"><i class="fas fa-clock"></i> ${s.duration}</div>
                    </div>
                    <div class="stop-time">${s.time}</div>
                </div>
            `).join('');

            // Show playback controls
            document.getElementById('playbackControls').style.display = 'flex';
            document.getElementById('playbackSlider').max = routePoints.length - 1;
            document.getElementById('playbackSlider').value = 0;
            currentPoint = 0;
        }

        function calculateDistance(points) {
            let dist = 0;
            for (let i = 1; i < points.length; i++) {
                const d = Math.sqrt(
                    Math.pow(points[i].lat - points[i-1].lat, 2) +
                    Math.pow(points[i].lng - points[i-1].lng, 2)
                );
                dist += d * 111;
            }
            return dist;
        }

        function renderTimeline(points) {
            const timeline = document.getElementById('timeline');
            const keyPoints = points.filter((p, i) => i === 0 || i === points.length - 1 || i % 10 === 0);

            timeline.innerHTML = keyPoints.map((p, i) => {
                const isStart = i === 0;
                const isEnd = i === keyPoints.length - 1;
                const dotClass = isStart ? 'start' : isEnd ? 'stop' : p.status === 'stop' ? 'idle' : 'checkpoint';

                return `
                    <div class="timeline-item">
                        <div class="timeline-dot ${dotClass}"></div>
                        <div class="timeline-content">
                            <div class="timeline-time">${p.time.toLocaleTimeString('en-IN', {hour:'2-digit',minute:'2-digit'})}</div>
                            <div class="timeline-location">${isStart ? 'Trip Started' : isEnd ? 'Trip Ended' : 'Checkpoint ' + i}</div>
                            <div class="timeline-detail">Speed: ${p.speed} km/h | Lat: ${p.lat.toFixed(4)}</div>
                        </div>
                    </div>
                `;
            }).join('');
        }

        function focusStop(lat, lng) {
            map.setView([lat, lng], 16);
        }

        function fitRoute() {
            if (routePolyline) {
                map.fitBounds(routePolyline.getBounds().pad(0.1));
            }
        }

        function toggleSatellite() {
            satelliteMode = !satelliteMode;
            map.eachLayer(layer => {
                if (layer instanceof L.TileLayer) map.removeLayer(layer);
            });

            if (satelliteMode) {
                L.tileLayer('https://server.arcgisonline.com/ArcGIS/rest/services/World_Imagery/MapServer/tile/{z}/{y}/{x}', {
                    attribution: 'Sarvam GPS',
                    maxZoom: 19
                }).addTo(map);
            } else {
                L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png', {
                    attribution: 'Sarvam GPS',
                    maxZoom: 19
                }).addTo(map);
            }
        }

        // ============ PLAYBACK ============
        function togglePlayback() {
            const btn = document.getElementById('playBtn');
            if (isPlaying) {
                clearInterval(playbackInterval);
                isPlaying = false;
                btn.innerHTML = '<i class="fas fa-play"></i>';
            } else {
                isPlaying = true;
                btn.innerHTML = '<i class="fas fa-pause"></i>';
                playbackInterval = setInterval(() => {
                    if (currentPoint >= routePoints.length - 1) {
                        resetPlayback();
                        return;
                    }
                    currentPoint++;
                    updatePlayback();
                }, 200 / playbackSpeed);
            }
        }

        function resetPlayback() {
            clearInterval(playbackInterval);
            isPlaying = false;
            currentPoint = 0;
            document.getElementById('playBtn').innerHTML = '<i class="fas fa-play"></i>';
            updatePlayback();
        }

        function updatePlayback() {
            const p = routePoints[currentPoint];
            movingMarker.setLatLng([p.lat, p.lng]);
            document.getElementById('playbackSlider').value = currentPoint;
            document.getElementById('playbackTime').textContent = p.time.toLocaleTimeString('en-IN', {hour:'2-digit',minute:'2-digit'});
        }

        function seekPlayback(value) {
            currentPoint = parseInt(value);
            updatePlayback();
        }

        function setSpeed(speed) {
            playbackSpeed = speed;
            document.querySelectorAll('.speed-btn').forEach(btn => btn.classList.remove('active'));
            event.target.classList.add('active');
            if (isPlaying) {
                clearInterval(playbackInterval);
                playbackInterval = setInterval(() => {
                    if (currentPoint >= routePoints.length - 1) {
                        resetPlayback();
                        return;
                    }
                    currentPoint++;
                    updatePlayback();
                }, 200 / playbackSpeed);
            }
        }

        function exportRoute() {
            showToast('Route exported to PDF!', 'success');
        }

        function showToast(message, type) {
            const container = document.getElementById('toastContainer');
            const toast = document.createElement('div');
            toast.className = `toast ${type}`;
            toast.innerHTML = `<i class="fas fa-${type === 'success' ? 'check-circle' : 'info-circle'}"></i> ${message}`;
            container.appendChild(toast);
            setTimeout(() => toast.classList.add('show'), 10);
            setTimeout(() => {
                toast.classList.remove('show');
                setTimeout(() => toast.remove(), 400);
            }, 3000);
        }

        // ============ INIT ============
        initMap();
    </script>
</body>
</html>
