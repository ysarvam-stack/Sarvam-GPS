        // ========== RENDER TIMELINE ==========
        function renderTimeline(data = timelineData) {
            const container = document.getElementById('timeline');
            container.innerHTML = '';

            data.forEach(item => {
                const div = document.createElement('div');
                div.className = 'timeline-item';
                
                const typeClass = item.type;
                const badgeClass = `badge-${item.type}`;
                const badgeText = item.type.toUpperCase();
                const durationPercent = item.type === 'stop' ? 30 : item.type === 'idle' ? 10 : 60;

                div.innerHTML = `
                    <div class="timeline-dot ${typeClass}"></div>
                    <div class="timeline-content">
                        <div class="timeline-header">
                            <div>
                                <h4>${item.title}</h4>
                                <span style="font-size: 12px; color: #888;">${item.time}</span>
                            </div>
                            <span class="timeline-badge ${badgeClass}">${badgeText}</span>
                        </div>
                        <div class="timeline-details">
                            <span><i class="fas fa-car"></i> ${item.vehicle}</span>
                            <span><i class="fas fa-user"></i> ${item.driver}</span>
                            <span><i class="fas fa-map-marker-alt"></i> ${item.location}</span>
                        </div>
                        <p style="font-size: 12px; color: #666; margin-top: 8px;">${item.details}</p>

                `;

            });
        }

        function filterTimeline() {
            const status = document.getElementById('statusFilter').value;
            if (status === 'all') {
                renderTimeline();
            } else {
                const filtered = timelineData.filter(t => t.type === status);
                renderTimeline(filtered);
            }
        }
